import LayerFormat from './Layer';
import VectorStyle from '../../layer/VectorStyle'
import VectorSource from 'ol/source/vector';
import Feature from 'ol/Feature';

import { ol_geom_createFromType } from 'ol-ext/geom/GeomUtils'

/**
 * @classdesc
 * Geoportail layer format reader/writer.
 * @api
 */
class VectorStyleFormat extends LayerFormat {
  /** Constructor 
   */
  constructor() {
    super();
  }
}

/** Read Geoportail layer
 * @param {options} json source
 * @return {VectorStyle}
 */
VectorStyleFormat.prototype.read = function (options) {
  if (options.type !== 'Vector') return;
  const source = new VectorSource();
  const layer = new VectorStyle({ source: source });
  this.readOptions(layer, options);
  layer.setIgnStyle(options.style || {}); 
  if (options.cluster) {
    layer.setMode('cluster', { disatance: options.radiusCluster });
    layer.set('maxZoomCluster', parseInt(options.maxZoomCluster));
  }
  layer.setPopupContent(options.popupContent);
  // Read features
  const features = [];
  options.features.forEach((f) => {
    const feature = new Feature(ol_geom_createFromType(f.type, f.coords));
    feature.setProperties(f.attributes);
    feature.setIgnStyle(f.style);
    feature.setPopupContent(f.popupcontent);
    features.push(feature);
  });
  source.addFeatures(features)
  return layer;
};

/** Write layer
 *	@param {VectorStyle}
 *	@return {object} json data
 */
VectorStyleFormat.prototype.write = function (layer) {
  if (!(layer instanceof VectorStyle)) return false;
  var s = this.writeOptions(layer, { type: 'dessin' });
  return s;
};

export default VectorStyleFormat

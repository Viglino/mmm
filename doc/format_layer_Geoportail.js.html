import { getUid } from 'ol/util';
import ol_Object from 'ol/Object'
import GeoportailLayer from 'ol-ext/layer/Geoportail'
import Filter from 'ol-ext/filter/CSS'
import LayerFormat from './Layer';

/**
 * @classdesc
 * Geoportail layer format reader/writer.
 * @api
 */
class GeoportailFormat extends LayerFormat {
  /** Constructor 
   */
  constructor() {
    super();
  }
}

/** Read Geoportail layer
 * @param {options} json source
 * @return {Geoportail}
 */
GeoportailFormat.prototype.read = function (options, key) {
  if (options.type !== 'Geoportail') return;
  // Old GPP services
  switch (options.layer) {
    case 'GEOGRAPHICALGRIDSYSTEMS.MAPS.SCAN-EXPRESS.NIVEAUXGRIS': {
      options.layer = 'GEOGRAPHICALGRIDSYSTEMS.PLANIGNV2';
      options.filter = 'grayscale'
      break;
    } 
    case 'GEOGRAPHICALGRIDSYSTEMS.PLANIGN': {
      options.layer = 'GEOGRAPHICALGRIDSYSTEMS.PLANIGNV2';
      break;
    }
  }
  // Create
  try {
    const layer = new GeoportailLayer({
      className: 'gpp_'+getUid(new ol_Object),
      gppKey: key,
      layer: options.layer
    });
    if (options.filter) {
      const filter = new Filter({ filter: 'grayscale(1)' });
      layer.addFilter(filter)
      window.filter = filter
    }
    this.readOptions(layer, options);
    return layer;
  } catch (e) {
    console.error(e.message);
    return false;
  }
};

/** Write Geoportail layer
 *	@param {ol.layer.Geoportail}
 *	@return {object} json data
 */
GeoportailFormat.prototype.write = function (layer) {
  if (!(layer instanceof GeoportailLayer)) return false;
  var s = this.writeOptions(layer, { type: 'Geoportail' });
  s.layer = layer.get('layer');
  return s;
};

export default GeoportailFormat
